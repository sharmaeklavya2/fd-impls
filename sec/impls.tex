\begin{tightenum}
\item[*] \label{foot:efx-to-ef1}These results hold for additional settings.
    See \cref{thm:impl:efx-to-ef1} in \cref{sec:impls-extra:among-ef-efx-ef1} for details.
\item[\textdagger] \label{foot:dbl-mono-1}A function $v: 2^M \to \mathbb{R}$ is \emph{doubly monotone}
    if $M = G \cup C$, and $\forall\,R \subseteq M$,
    we have $v(g \mid R) \ge 0$ for all $g \in G \setminus R$
    and $v(c \mid R) \le 0$ for all $c \in C \setminus R$.
\item[\textdaggerdbl] \label{foot:pg}In addition to $F_1 \fimplies F_2$,
    we also get p-$F_1$ $\fimplies$ p-$F_2$ and g-$F_1$ $\fimplies$ g-$F_2$.
\item[\S] \label{foot:eef1-to-prop1}\cite{aziz2021fair} proved this for additive valuations.
    Recently, \cite{andersen2025computing} proved this independently for submodular valuations.
\end{tightenum}
